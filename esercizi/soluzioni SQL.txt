1) SELECT *
	FROM clienti
	WHERE nome = 'Mario'
2) SELECT regione
	FROM regioni
	WHERE area_geografica = 'sud'
3) SELECT *
	FROM prenotazioni
	WHERE importo > 200
4) SELECT *
	FROM prenotazioni
	WHERE importo > 150
	AND caparra < 50
5) --errore nel testo, l'esercizio non poteva essere risolto quando l'ho assegnato--
	SELECT DISTINCT regione
	FROM regioni
	INNER JOIN citta ON regioni.id_regione = citta.regione
	INNER JOIN clienti ON citta.id_citta = clienti.citta
	INNER JOIN prenotazioni ON clienti.id_cliente = prenotazioni.cliente
	WHERE importo > 150 AND caparra < 50
6) SELECT DISTINCT nome, cognome
	FROM clienti
	ORDER BY dataNascita ASC
7) SELECT DISTINCT arrivo
	FROM prenotazioni
	WHERE tipo_struttura='2 Stelle'
	ORDER BY arrivo DESC
8) SELECT DISTINCT clienti.nome, clienti.cognome
	FROM clienti 
	INNER JOIN prenotazioni ON clienti.ID_cliente = prenotazioni.cliente
	WHERE  prenotazioni.arrivo < '2015-07-01'
9) SELECT DISTINCT citta.citta, nome, cognome
	FROM clienti 
	INNER JOIN prenotazioni ON clienti.ID_cliente = prenotazioni.cliente 
	INNER JOIN citta ON citta.id_citta = clienti.citta
	WHERE prenotazioni.caparra < 50
	AND prenotazioni.importo > 150
